\newcommand{\cvInvalidMonth}{\emph{\textbf{invalid-month}}}

\newcommand{\cvMonth}[1]{%
    \ifcase#1 \cvInvalidMonth%
    \or Jan%
    \or Feb%
    \or Mar%
    \or Apr%
    \or May%
    \or Jun%
    \or Jul%
    \or Aug%
    \or Sep%
    \or Oct%
    \or Nov%
    \or Dec%
    \else
        \cvInvalidMonth%
    \fi
}

\newcommand{\cvDate}[2]{%
    \emph{\cvMonth{#1}} {#2}%
}
